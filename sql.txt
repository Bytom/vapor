CREATE TABLE `deals` (`id` int(11) NOT NULL AUTO_INCREMENT,  `symbol_id` int(11) NOT NULL,  `side` tinyint(1) NOT NULL,  `low` decimal(20,10) DEFAULT '0.0000000000',  `high` decimal(20,10) DEFAULT '0.0000000000',  `latest` decimal(20,10) DEFAULT '0.0000000000',  `amount` bigint(20) DEFAULT '0',  `duration_minutes` int(4) DEFAULT '1',  `end_timestamp` bigint(13) NOT NULL,  `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,  `updated_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,  PRIMARY KEY (`id`),  KEY `symbol_id` (`symbol_id`)) ENGINE=InnoDB DEFAULT CHARSET=utf8;
